146  vector3f p = gameObject->Transform()->Position(); // Camera position
147  vector3f up = gameObject->Transform()->up(); // Up vector (for "screen rotation")

150  if(viewmode == Viewmode::ortho or viewmode == Viewmode::ortho_screen) // Añadir rotaciones o algo
151  return;

153  // Solución temporal: usar un pivote anclado al objeto que manipule la cámara como un hijo.
154  // Mientras la cámara rote, se aplicarán las rotaciones al pivote, lo que permitirá a la cámara mirar libremente.
155  // -> La solución es un poco burda, pero sencilla. Al menos, no complicamos las cosas con procesos matemáticos innecesarios.
156  //
157  // Nota: Recalcular la posición global del pivote o del objetivo en cada iteración, para cada cámara,
158  // puede resultar pesado si la cámara está anidada a otros objetos. Sería conveniente declarar un
159  // previous_pos y un pos para evitar recalcular si son iguales (ganamos GPU y CPU a cambio de 48 bytes).
160 
161  CGameObject* target_p = gSystem_GameObject_Manager[target];
162 
163  if(!target_p) // Añadir pivote y calcular su posición global:
164  tp = pivot->Transform()->Position(); // Inservible...
165  else // O coger la posición global del objeto.
166  tp = target_p->Transform()->Position();
167 
168  // ->xBUG En CComponent_Camera::Setup() Caso fatal; hay que tomar un valor válido del vector "up"
169  //if((p - tp).normalize() == up or (p - tp).normalize() == vector3f())
170  //up.x += 0.05f;

172  modelViewMatrix = glm::lookAt(p.to_glm(), tp.to_glm(), up.to_glm());
