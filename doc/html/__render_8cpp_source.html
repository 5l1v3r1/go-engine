678  // Idea sencilla: dibujamos un cubo de tamaño 1x1 sin depth_test justo donde esta la cámara.

695  glUniformMatrix4fv(simpleShader->GetUniformIndex("ModelViewMatrix") , 1, GL_FALSE, glm::value_ptr(local_modelViewMatrix));
696  glUniform1i(simpleShader->GetUniformIndex("texture"), 0);
697  glUniform1f(simpleShader->GetUniformIndex("textureFlag"), 1.0f);
698  glUniform4f(simpleShader->GetUniformIndex("in_Color"), 1.0, 1.0, 1.0, 1.0);
